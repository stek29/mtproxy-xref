21#include <assert.h>
22#include <cpuid.h>
23
24#include "cpuid.h"
25
26
27#define CPUID_MAGIC 0x280147b8
28
29kdb_cpuid_t *kdb_cpuid (void) {
30 static kdb_cpuid_t cached = { .magic = 0 };
31 if (cached.magic) {
32 assert (cached.magic == CPUID_MAGIC);
33 return &cached;
34 }
35
36 unsigned int a;
37 assert(
38 __get_cpuid(1,
39 &a,
40 (unsigned int*) &cached.ebx,
41 (unsigned int*) &cached.ecx,
42 (unsigned int*) &cached.edx
43 ) != 0
44 );
45
46 cached.magic = CPUID_MAGIC;
47 return &cached;
48}
