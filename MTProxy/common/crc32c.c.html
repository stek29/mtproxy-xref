565 const int *p = (const int *) data;
566 int x;
567#define DO_ONE(v) crc ^= v; crc = crc32c_table0[crc & 0xff] ^ crc32c_table1[(crc & 0xff00) >> 8] ^ crc32c_table2[(crc & 0xff0000) >> 16] ^ crc32c_table[crc >> 24];

571 DO_FOUR (p);
572 DO_FOUR (p + 4);

577 p += 4;

581 DO_ONE (p[1]);

617static unsigned crc32c_combine_clmul (unsigned crc1, unsigned crc2, int64_t len2) {
618 static unsigned int crc32c_powers[252] __attribute__ ((aligned(16)));
619 if (len2 <= 0) {

621 }
622 if (!crc32c_powers[251]) {
623 gf32_compute_powers_clmul (crc32c_powers, CRC32C_REFLECTED_POLY);

629 assert (crc32c_powers[251]);

632 uint64_t T = gf32_combine_clmul (crc32c_powers, crc1, len2);

638 : "0" (crc), "r" ( (unsigned) T)
639 );
