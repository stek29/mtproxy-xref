105static int tcp_rpcc_process_nonce_packet (connection_job_t C, struct raw_message *msg) /* {{{ */ {
106 struct connection_info *c = CONN_INFO (C);
107
108 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
109 union {
110 struct tcp_rpc_nonce_packet s;
111 struct tcp_rpc_nonce_ext_packet x;
112 struct tcp_rpc_nonce_dh_packet dh;
113 } P;
114 struct tcp_rpc_nonce_dh_packet *dh = 0;
115 int res;
116
117 unsigned char temp_dh[256];
118 int temp_dh_len = 0;
119
120 int packet_num = D->in_packet_num - 1;
121 int packet_type;
122 assert (rwm_fetch_lookup (msg, &packet_type, 4) == 4);
123 int packet_len = msg->total_bytes;
124
125 if (packet_num != -2 || packet_type != RPC_NONCE) {
126 return -2;
127 }
128
129 if (packet_len < sizeof (struct tcp_rpc_nonce_packet) || packet_len > sizeof (struct tcp_rpc_nonce_dh_packet)) {
130 return -3;
131 }
132
133 assert (rwm_fetch_data (msg, &P, packet_len) == packet_len);
134
135 vkprintf (2, "Processing nonce packet, crypto schema: %d, key select: %d\n", P.s.crypto_schema, P.s.key_select);
136
137 switch (P.s.crypto_schema) {
138 case RPC_CRYPTO_NONE:
139 if (packet_len != sizeof (struct tcp_rpc_nonce_packet)) {
140 return -3;
141 }
142 break;
143 case RPC_CRYPTO_AES:
144 if (packet_len != sizeof (struct tcp_rpc_nonce_packet)) {
145 return -3;
146 }
147 break;
148 case RPC_CRYPTO_AES_EXT:
149 if (packet_len < sizeof (struct tcp_rpc_nonce_ext_packet) - 4 * RPC_MAX_EXTRA_KEYS) {
150 return -3;
151 }
152 if (P.x.extra_keys_count < 0 || P.x.extra_keys_count > RPC_MAX_EXTRA_KEYS || packet_len != sizeof (struct tcp_rpc_nonce_ext_packet) + 4*(P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS)) {
153 return -3;
154 }
155 break;
156 case RPC_CRYPTO_AES_DH:
157 if (packet_len < sizeof (struct tcp_rpc_nonce_dh_packet) - 4 * RPC_MAX_EXTRA_KEYS) {
158 return -3;
159 }
160 if (P.x.extra_keys_count < 0 || P.x.extra_keys_count > RPC_MAX_EXTRA_KEYS || packet_len != sizeof (struct tcp_rpc_nonce_dh_packet) + 4 * (P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS)) {
161 return -3;
162 }
163 break;
164 default:
165 return -3;
166 }
167
168 switch (P.s.crypto_schema) {
169 case RPC_CRYPTO_NONE:
170 if (P.s.key_select) {
171 return -3;
172 }
173 if (D->crypto_flags & RPCF_ALLOW_UNENC) {
174 if (D->crypto_flags & RPCF_ALLOW_ENC) {
175 // release_all_unprocessed (&c->Out);
176 assert (!c->out_p.total_bytes);
177 }
178 D->crypto_flags = RPCF_ALLOW_UNENC;
179 } else {
180 return -5;
181 }
182 break;
183 case RPC_CRYPTO_AES_DH: {
184 dh = (struct tcp_rpc_nonce_dh_packet *)((char *) &P + 4 * (P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS));
185 if (!dh_params_select) {
186 init_dh_params ();
187 }
188 if (!dh->dh_params_select || dh->dh_params_select != dh_params_select) {
189 return -7;
190 }
191 if (!(D->crypto_flags & RPCF_REQ_DH) || !c->crypto_temp) {
192 return -7;
193 }
194 }
195 case RPC_CRYPTO_AES_EXT:
196 P.s.key_select = select_best_key_signature (P.s.key_select, P.x.extra_keys_count, P.x.extra_key_select);
197 case RPC_CRYPTO_AES:
198 if (!P.s.key_select || !select_best_key_signature (P.s.key_select, 0, 0)) {
199 return -3;
200 }
201 if (!(D->crypto_flags & RPCF_ALLOW_ENC)) {
202 return -5;
203 }
204 if (abs (P.s.crypto_ts - D->nonce_time) > 30) {
205 return -6;
206 }
207 if ((D->crypto_flags & (RPCF_REQ_DH | RPCF_ALLOW_SKIP_DH)) == RPCF_REQ_DH && !dh) {
208 return -7;
209 }
210 if (dh) {
211 temp_dh_len = dh_third_round (temp_dh, dh->g_a, c->crypto_temp);
212 if (temp_dh_len != 256) {
213 return -8;
214 }
215 //active_dh_connections++;
216 incr_active_dh_connections ();
217 __sync_fetch_and_or (&c->flags, C_ISDH);
218 }
219 if (c->crypto_temp) {
220 if (((struct crypto_temp_dh_params *) c->crypto_temp)->magic == CRYPTO_TEMP_DH_PARAMS_MAGIC) {
221 free_crypto_temp (c->crypto_temp, sizeof (struct crypto_temp_dh_params));
222 } else {
223 free_crypto_temp (c->crypto_temp, 0);
224 }
225 c->crypto_temp = 0;
226 }
227 res = TCP_RPCC_FUNC(C)->rpc_start_crypto (C, P.s.crypto_nonce, P.s.key_select, temp_dh, temp_dh_len);
228 if (res < 0) {
229 return -6;
230 }
231 break;
232 default:
233 return -4;
234 }
235
236 vkprintf (2, "Processed nonce packet, crypto flags = %d\n", D->crypto_flags);
237 return 0;
238}

241static int tcp_rpcc_send_handshake_packet (connection_job_t C) /* {{{ */ {
242 struct connection_info *c = CONN_INFO (C);
243
244 struct tcp_rpc_data *D = TCP_RPC_DATA (C);
245 struct tcp_rpc_handshake_packet P;
246 assert (PID.ip);
247 memset (&P, 0, sizeof (P));
248 P.type = RPC_HANDSHAKE;
249 P.flags = tcp_get_default_rpc_flags () & RPCF_USE_CRC32C;
250 if (!D->remote_pid.port) {
251 D->remote_pid.ip = (c->remote_ip == 0x7f000001 ? 0 : c->remote_ip);
252 D->remote_pid.port = c->remote_port;
253 }
254 memcpy (&P.sender_pid, &PID, sizeof (struct process_id));
255 memcpy (&P.peer_pid, &D->remote_pid, sizeof (struct process_id));
256
257 tcp_rpc_conn_send_data (JOB_REF_CREATE_PASS (C), sizeof (P), &P);
258
259 return 0;
260}

280static int tcp_rpcc_process_handshake_packet (connection_job_t C, struct raw_message *msg) /* {{{ */ {
281 //struct connection_info *c = CONN_INFO (C);
282
283 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
284 struct tcp_rpc_handshake_packet P;
285
286 int packet_num = D->in_packet_num - 1;
287 int packet_len = msg->total_bytes;
288 int packet_type;
289 assert (rwm_fetch_lookup (msg, &packet_type, 4) == 4);
290
291 if (packet_num != -1 || packet_type != RPC_HANDSHAKE) {
292 return -2;
293 }
294 if (packet_len != sizeof (struct tcp_rpc_handshake_packet)) {
295 tcp_rpcc_send_handshake_error_packet (C, -3);
296 return -3;
297 }
298 assert (rwm_fetch_data (msg, &P, packet_len) == packet_len);
299 if (!matches_pid (&P.sender_pid, &D->remote_pid) && !(TCP_RPCC_FUNC(C)->mode_flags & TCP_RPC_IGNORE_PID)) {
300 vkprintf (1, "PID mismatch during client RPC handshake: local %08x:%d:%d:%d, remote %08x:%d:%d:%d\n",
301 D->remote_pid.ip, D->remote_pid.port, D->remote_pid.pid, D->remote_pid.utime, P.sender_pid.ip, P.sender_pid.port, P.sender_pid.pid, P.sender_pid.utime);
302 tcp_rpcc_send_handshake_error_packet (C, -6);
303 return -6;
304 }
305 if (!P.sender_pid.ip) {
306 P.sender_pid.ip = D->remote_pid.ip;
307 }
308 memcpy (&D->remote_pid, &P.sender_pid, sizeof (struct process_id));
309 if (!matches_pid (&PID, &P.peer_pid)) {
310 tcp_rpcc_send_handshake_error_packet (C, -4);
311 return -4;
312 }
313 if (P.flags & 0xff) {
314 tcp_rpcc_send_handshake_error_packet (C, -7);
315 return -7;
316 }
317 if (P.flags & RPCF_USE_CRC32C) {
318 if (!(tcp_get_default_rpc_flags () & RPCF_USE_CRC32C)) {
319 tcp_rpcc_send_handshake_error_packet (C, -8);
320 return -8;
321 }
322 D->crypto_flags |= RPCF_USE_CRC32C;
323 D->custom_crc_partial = crc32c_partial;
324 }
325 return 0;
326}

329int tcp_rpcc_parse_execute (connection_job_t C) /* {{{ */ {
330 struct connection_info *c = CONN_INFO (C);
331
332 vkprintf (4, "%s. in_total_bytes = %d\n", __func__, c->in.total_bytes);
333 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
334 int len;
335
336 while (1) {
337 len = c->in.total_bytes;
338 if (len <= 0) {
339 break;
340 }
341 if (len < 4) {
342 return 4 - len;
343 }
344
345 int packet_len;
346 assert (rwm_fetch_lookup (&c->in, &packet_len, 4) == 4);
347 if (packet_len <= 0 || (packet_len & 3) || (packet_len > TCP_RPCC_FUNC(C)->max_packet_len && TCP_RPCC_FUNC(C)->max_packet_len > 0)) {
348 vkprintf (1, "error while parsing packet: bad packet length %d\n", packet_len);
349 fail_connection (C, -1);
350 return 0;
351 }
352
353 if (packet_len == 4) {
354 assert (rwm_skip_data (&c->in, 4) == 4);
355 continue;
356 }
357
358 if (packet_len < 16) {
359 vkprintf (1, "error while parsing packet: bad packet length %d\n", packet_len);
360 fail_connection (C, -2);
361 return 0;
362 }
363
364 if (len < packet_len) {
365 return packet_len - len;
366 }
367
368
369 struct raw_message msg;
370 if (c->in.total_bytes == packet_len) {
371 msg = c->in;
372 rwm_init (&c->in, 0);
373 } else {
374 rwm_split_head (&msg, &c->in, packet_len);
375 }
376
377 unsigned crc32;
378 assert (rwm_fetch_data_back (&msg, &crc32, 4) == 4);
379
380 unsigned packet_crc32 = rwm_custom_crc32 (&msg, packet_len - 4, D->custom_crc_partial);
381 if (crc32 != packet_crc32) {
382 vkprintf (1, "error while parsing packet: crc32 mismatch: %08x != %08x\n", packet_crc32, crc32);
383 fail_connection (C, -3);
384 rwm_free (&msg);
385 return 0;
386 }
387
388 assert (rwm_skip_data (&msg, 4) == 4); // len
389 int packet_num;
390 int packet_type;
391 assert (rwm_fetch_data (&msg, &packet_num, 4) == 4);
392 assert (rwm_fetch_lookup (&msg, &packet_type, 4) == 4);
393 packet_len -= 12;
394
395 if (verbosity > 2) {
396 fprintf (stderr, "received packet from connection %d\n", c->fd);
397 rwm_dump (&msg);
398 }
399
400 if (packet_num != D->in_packet_num) {
401 vkprintf (1, "error while parsing packet: got packet num %d, expected %d\n", packet_num, D->in_packet_num);
402 fail_connection (C, -4);
403 rwm_free (&msg);
404 return 0;
405 }
406
407 if (packet_num < 0) {
408 D->in_packet_num ++;
409 int res;
410 if (packet_num == -2) {
411 res = tcp_rpcc_process_nonce_packet (C, &msg);
412 if (res >= 0) {
413 res = tcp_rpcc_send_handshake_packet (C);
414 }
415 } else if (packet_num == -1) {
416 res = tcp_rpcc_process_handshake_packet (C, &msg);
417 if (res >= 0 && TCP_RPCC_FUNC(C)->rpc_ready) {
418 notification_event_insert_tcp_conn_ready (C);
419 }
420 } else {
421 vkprintf (1, "bad packet num %d\n", packet_num);
422 res = -5;
423 }
424
425 rwm_free (&msg);
426 if (res < 0) {
427 fail_connection (C, res);
428 return 0;
429 }
430 continue;
431 }
432
433 D->in_packet_num ++;
434
435 int res;
436 if (packet_type == RPC_PING) {
437 res = tcp_rpc_default_execute (C, packet_type, &msg);
438 } else {
439 res = TCP_RPCC_FUNC(C)->execute (C, packet_type, &msg);
440 }
441
442 if (res <= 0) {
443 rwm_free (&msg);
444 }
445 }
446 return 0;
447}

450int tcp_rpcc_connected (connection_job_t C) /* {{{ */ {
451 struct connection_info *c = CONN_INFO (C);
452
453 TCP_RPC_DATA(C)->out_packet_num = -2;
454 c->last_query_sent_time = precise_now;
455
456 if (TCP_RPCC_FUNC(C)->rpc_check_perm) {
457 int res = TCP_RPCC_FUNC(C)->rpc_check_perm (C);
458 if (res < 0) {
459 return res;
460 }
461 res &= RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC | RPCF_REQ_DH | RPCF_ALLOW_SKIP_DH;
462 if (!(res & (RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC))) {
463 return -1;
464 }
465 TCP_RPC_DATA(C)->crypto_flags = res;
466 } else {
467 TCP_RPC_DATA(C)->crypto_flags = RPCF_ALLOW_ENC | RPCF_ALLOW_UNENC;
468 }
469 vkprintf (2, "RPC connection #%d: [%s]:%d -> [%s]:%d connected, crypto_flags = %d\n", c->fd, show_our_ip (C), c->our_port, show_remote_ip (C), c->remote_port, TCP_RPC_DATA(C)->crypto_flags);
470
471 assert (TCP_RPCC_FUNC(C)->rpc_init_crypto);
472 int res = TCP_RPCC_FUNC(C)->rpc_init_crypto (C);
473
474 if (res > 0) {
475 assert (TCP_RPC_DATA(C)->crypto_flags & RPCF_ENC_SENT);
476 } else {
477 return -1;
478 }
479
480 assert (TCP_RPCC_FUNC(C)->flush_packet);
481 TCP_RPCC_FUNC(C)->flush_packet (C);
482
483 return 0;
484}

500int tcp_rpcc_default_check_ready (connection_job_t C) {
501 struct connection_info *c = CONN_INFO (C);
502
503 if (c->flags & C_ERROR) {
504 return c->ready = cr_failed;
505 }
506
507 const double CONNECT_TIMEOUT = 3.0;
508 if (c->status == conn_connecting || TCP_RPC_DATA(C)->in_packet_num < 0) {
509 //if (TCP_RPC_DATA(C)->in_packet_num == -1 && c->status == conn_running) {
510 // return c->ready = cr_ok;
511 //}
512
513 assert (c->last_query_sent_time != 0);
514 if (c->last_query_sent_time < precise_now - CONNECT_TIMEOUT) {
515 fail_connection (C, -6);
516 return c->ready = cr_failed;
517 }
518 return c->ready = cr_notyet;
519 }
520
521 if (c->status == conn_working) {
522 return c->ready = cr_ok;
523 }
524
525 fail_connection (C, -7);
526 return c->ready = cr_failed;
527}

549int tcp_rpcc_init_outbound (connection_job_t C) {
550 struct connection_info *c = CONN_INFO (C);
551
552 vkprintf (3, "rpcc_init_outbound (%d)\n", c->fd);
553 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
554 c->last_query_sent_time = precise_now;
555 D->custom_crc_partial = crc32_partial;
556
557 if (TCP_RPCC_FUNC(C)->rpc_check_perm) {
558 int res = TCP_RPCC_FUNC(C)->rpc_check_perm (C);
559 if (res < 0) {
560 return res;
561 }
562 res &= RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC | RPCF_REQ_DH | RPCF_ALLOW_SKIP_DH;
563 if (!(res & (RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC))) {
564 return -1;
565 }
566 if (res & RPCF_REQ_DH) {
567 if (tcp_add_dh_accept () < 0) {
568 return -1;
569 }
570 }
571
572 D->crypto_flags = res;
573 } else {
574 D->crypto_flags = RPCF_ALLOW_UNENC;
575 }
576
577 D->in_packet_num = -2;
578
579 return 0;
580}

596int tcp_rpcc_init_crypto (connection_job_t C) {
597 struct connection_info *c = CONN_INFO (C);
598
599 if (!(TCP_RPC_DATA(C)->crypto_flags & RPCF_ALLOW_ENC)) {
600 return tcp_rpcc_init_fake_crypto (C);
601 }
602
603 TCP_RPC_DATA(C)->nonce_time = time (0);
604
605 aes_generate_nonce (TCP_RPC_DATA(C)->nonce);
606
607 if (!dh_params_select) {
608 assert (init_dh_params () >= 0);
609 assert (dh_params_select);
610 }
611
612 union {
613 struct tcp_rpc_nonce_packet s;
614 struct tcp_rpc_nonce_ext_packet x;
615 struct tcp_rpc_nonce_dh_packet dh;
616 } buf;
617 int len = sizeof (struct tcp_rpc_nonce_packet);
618
619 memset (&buf, 0, sizeof (buf));
620 memcpy (buf.s.crypto_nonce, TCP_RPC_DATA(C)->nonce, 16);
621 buf.s.crypto_ts = TCP_RPC_DATA(C)->nonce_time;
622 buf.s.type = RPC_NONCE;
623 buf.s.key_select = main_secret.key_signature;
624 buf.s.crypto_schema = RPC_CRYPTO_AES;
625 int extra_keys = buf.x.extra_keys_count = 0;
626 assert (extra_keys >= 0 && extra_keys <= RPC_MAX_EXTRA_KEYS);
627
628 if (TCP_RPC_DATA(C)->crypto_flags & RPCF_REQ_DH) {
629 buf.s.crypto_schema = RPC_CRYPTO_AES_DH;
630 len = sizeof (struct tcp_rpc_nonce_dh_packet) + 4*(extra_keys - RPC_MAX_EXTRA_KEYS);
631 struct tcp_rpc_nonce_dh_packet *dh = (struct tcp_rpc_nonce_dh_packet *)((char *) &buf + 4*(extra_keys - RPC_MAX_EXTRA_KEYS));
632 dh->dh_params_select = dh_params_select;
633 assert (!c->crypto_temp);
634 c->crypto_temp = alloc_crypto_temp (sizeof (struct crypto_temp_dh_params));
635 assert (c->crypto_temp);
636 dh_first_round (dh->g_a, c->crypto_temp);
637 } else if (extra_keys) {
638 buf.s.crypto_schema = RPC_CRYPTO_AES_EXT;
639 len = offsetof (struct tcp_rpc_nonce_ext_packet, extra_key_select) + 4 * extra_keys;
640 }
641
642 tcp_rpc_conn_send_data (JOB_REF_CREATE_PASS (C), len, &buf);
643
644 assert ((TCP_RPC_DATA(C)->crypto_flags & (RPCF_ALLOW_ENC | RPCF_ENC_SENT)) == RPCF_ALLOW_ENC);
645 TCP_RPC_DATA(C)->crypto_flags |= RPCF_ENC_SENT;
646
647 assert (!c->crypto);
648
649 return 1;
650}

652int tcp_rpcc_start_crypto (connection_job_t C, char *nonce, int key_select, unsigned char *temp_key, int temp_key_len) {
653 struct connection_info *c = CONN_INFO (C);
654
655 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
656
657 vkprintf (2, "rpcc_start_crypto: key_select = %d\n", key_select);
658
659 if (c->crypto) {
660 return -1;
661 }
662
663 if (c->in.total_bytes || c->out.total_bytes || !D->nonce_time) {
664 return -1;
665 }
666
667 if (!key_select) {
668 return -1;
669 }
670
671 aes_secret_t *secret = &main_secret;
672
673 struct aes_key_data aes_keys;
674
675 if (aes_create_keys (&aes_keys, 1, nonce, D->nonce, D->nonce_time, nat_translate_ip (c->remote_ip), c->remote_port, c->remote_ipv6, nat_translate_ip (c->our_ip), c->our_port, c->our_ipv6, secret, temp_key, temp_key_len) < 0) {
676 return -1;
677 }
678
679 if (aes_crypto_init (C, &aes_keys, sizeof (aes_keys)) < 0) {
680 return -1;
681 }
682
683 return 1;
684}
