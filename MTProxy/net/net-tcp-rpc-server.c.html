100int tcp_rpcs_default_execute (connection_job_t C, int op, struct raw_message *raw) {
101 struct connection_info *c = CONN_INFO (C);
102
103 vkprintf (3, "%s: fd=%d, op=%d, len=%d\n", __func__, c->fd, op, raw->total_bytes);
104 if (op == RPC_PING && raw->total_bytes == 12) {
105 c->last_response_time = precise_now;
106 int P[3];
107 assert (rwm_fetch_data (raw, P, 12) == 12);
108 P[0] = RPC_PONG;
109 vkprintf (3, "received ping from " IP_PRINT_STR ":%d (val = %lld)\n", IP_TO_PRINT (c->remote_ip), (int)c->remote_port, *(long long *)(P + 1));
110 tcp_rpc_conn_send_data (JOB_REF_CREATE_PASS (C), 12, P);
111 return 0;
112 }
113 return 0;
114}

116static int tcp_rpcs_process_nonce_packet (connection_job_t C, struct raw_message *msg) {
117 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
118 union {
119 struct tcp_rpc_nonce_packet s;
120 struct tcp_rpc_nonce_ext_packet x;
121 struct tcp_rpc_nonce_dh_packet dh;
122 } P;
123 struct tcp_rpc_nonce_dh_packet *dh = 0;
124 int res;
125
126 int packet_num = D->in_packet_num;
127 int packet_type;
128 assert (rwm_fetch_lookup (msg, &packet_type, 4) == 4);
129 int packet_len = msg->total_bytes;
130
131 if (packet_num != -2 || packet_type != RPC_NONCE) {
132 return -2;
133 }
134 if (packet_len < sizeof (struct tcp_rpc_nonce_packet) || packet_len > sizeof (struct tcp_rpc_nonce_dh_packet)) {
135 return -3;
136 }
137
138 assert (rwm_fetch_data (msg, &P, packet_len) == packet_len);
139
140 switch (P.s.crypto_schema) {
141 case RPC_CRYPTO_NONE:
142 if (packet_len != sizeof (struct tcp_rpc_nonce_packet)) {
143 return -3;
144 }
145 break;
146 case RPC_CRYPTO_AES:
147 if (packet_len != sizeof (struct tcp_rpc_nonce_packet)) {
148 return -3;
149 }
150 break;
151 case RPC_CRYPTO_AES_EXT:
152 if (packet_len < sizeof (struct tcp_rpc_nonce_ext_packet) - 4 * RPC_MAX_EXTRA_KEYS) {
153 return -3;
154 }
155 if (P.x.extra_keys_count < 0 || P.x.extra_keys_count > RPC_MAX_EXTRA_KEYS || packet_len != sizeof (struct tcp_rpc_nonce_ext_packet) + 4 * (P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS)) {
156 return -3;
157 }
158 break;
159 case RPC_CRYPTO_AES_DH:
160 if (packet_len < sizeof (struct tcp_rpc_nonce_dh_packet) - 4 * RPC_MAX_EXTRA_KEYS) {
161 return -3;
162 }
163 if (P.x.extra_keys_count < 0 || P.x.extra_keys_count > RPC_MAX_EXTRA_KEYS || packet_len != sizeof (struct tcp_rpc_nonce_dh_packet) + 4 * (P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS)) {
164 return -3;
165 }
166 break;
167 default:
168 return -3;
169 }
170
171 switch (P.s.crypto_schema) {
172 case RPC_CRYPTO_NONE:
173 if (P.s.key_select) {
174 return -3;
175 }
176 if (D->crypto_flags & RPCF_ALLOW_UNENC) {
177 D->crypto_flags = RPCF_ALLOW_UNENC;
178 } else {
179 return -5;
180 }
181 break;
182 case RPC_CRYPTO_AES_DH: {
183 dh = (struct tcp_rpc_nonce_dh_packet *)((char *) &P + 4*(P.x.extra_keys_count - RPC_MAX_EXTRA_KEYS));
184 if (!dh_params_select) {
185 init_dh_params ();
186 }
187 if (!dh->dh_params_select || dh->dh_params_select != dh_params_select) {
188 dh = 0;
189 }
190 }
191 case RPC_CRYPTO_AES_EXT:
192 P.s.key_select = select_best_key_signature (P.s.key_select, P.x.extra_keys_count, P.x.extra_key_select);
193 case RPC_CRYPTO_AES:
194 if (!P.s.key_select || !select_best_key_signature (P.s.key_select, 0, 0)) {
195 if (D->crypto_flags & RPCF_ALLOW_UNENC) {
196 D->crypto_flags = RPCF_ALLOW_UNENC;
197 break;
198 }
199 return -3;
200 }
201 if (!(D->crypto_flags & RPCF_ALLOW_ENC)) {
202 if (D->crypto_flags & RPCF_ALLOW_UNENC) {
203 D->crypto_flags = RPCF_ALLOW_UNENC;
204 break;
205 }
206 return -5;
207 }
208 D->nonce_time = (now ? now : time (0));
209 if (abs (P.s.crypto_ts - D->nonce_time) > 30) {
210 return -6; //less'om
211 }
212 D->crypto_flags &= ~RPCF_ALLOW_UNENC;
213 break;
214 default:
215 if (D->crypto_flags & RPCF_ALLOW_UNENC) {
216 D->crypto_flags = RPCF_ALLOW_UNENC;
217 break;
218 }
219 return -4;
220 }
221
222 if ((D->crypto_flags & (RPCF_REQ_DH | RPCF_ALLOW_ENC)) == (RPCF_REQ_DH | RPCF_ALLOW_ENC) && !dh) {
223 if (D->crypto_flags & RPCF_ALLOW_SKIP_DH) {
224 D->crypto_flags &= ~(RPCF_REQ_DH | RPCF_ALLOW_SKIP_DH);
225 } else {
226 return -7;
227 }
228 }
229
230 res = TCP_RPCS_FUNC(C)->rpc_init_crypto (C, &P.s);
231 if (res < 0) {
232 return -6;
233 }
234 return 0;
235}

237static int tcp_rpcs_send_handshake_packet (connection_job_t c) {
238 struct tcp_rpc_data *D = TCP_RPC_DATA(c);
239 struct tcp_rpc_handshake_packet P;
240 assert (PID.pid);
241 memset (&P, 0, sizeof (P));
242 P.type = RPC_HANDSHAKE;
243 P.flags = D->crypto_flags & RPCF_USE_CRC32C;
244 memcpy (&P.sender_pid, &PID, sizeof (struct process_id));
245 memcpy (&P.peer_pid, &D->remote_pid, sizeof (struct process_id));
246
247 tcp_rpc_conn_send_data_im (JOB_REF_CREATE_PASS (c), sizeof (P), &P);
248 return 0;
249}

263static int tcp_rpcs_process_handshake_packet (connection_job_t C, struct raw_message *msg) {
264 struct connection_info *c = CONN_INFO (C);
265
266 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
267 struct tcp_rpc_handshake_packet P;
268 if (!PID.ip) {
269 init_server_PID (c->our_ip, c->our_port);
270 if (!PID.ip) {
271 PID.ip = get_my_ipv4 ();
272 }
273 }
274
275 int packet_num = D->in_packet_num;
276 int packet_type;
277 assert (rwm_fetch_lookup (msg, &packet_type, 4) == 4);
278 int packet_len = msg->total_bytes;
279
280 if (packet_num != -1 || packet_type != RPC_HANDSHAKE) {
281 return -2;
282 }
283 if (packet_len != sizeof (struct tcp_rpc_handshake_packet)) {
284 tcp_rpcs_send_handshake_error_packet (C, -3);
285 return -3;
286 }
287 assert (rwm_fetch_data (msg, &P, packet_len) == packet_len);
288 memcpy (&D->remote_pid, &P.sender_pid, sizeof (struct process_id));
289 if (!matches_pid (&PID, &P.peer_pid) && !(TCP_RPCS_FUNC(C)->mode_flags & TCP_RPC_IGNORE_PID)) {
290 vkprintf (1, "PID mismatch during handshake: local %08x:%d:%d:%d, remote %08x:%d:%d:%d\n",
291 PID.ip, PID.port, PID.pid, PID.utime, P.peer_pid.ip, P.peer_pid.port, P.peer_pid.pid, P.peer_pid.utime);
292 tcp_rpcs_send_handshake_error_packet (C, -4);
293 return -4;
294 }
295 if (P.flags & 0xff) {
296 tcp_rpcs_send_handshake_error_packet (C, -7);
297 return -7;
298 }
299 if (P.flags & tcp_get_default_rpc_flags () & RPCF_USE_CRC32C) {
300 D->crypto_flags |= RPCF_USE_CRC32C;
301 }
302 return 0;
303}

305int tcp_rpcs_parse_execute (connection_job_t C) {
306 struct connection_info *c = CONN_INFO (C);
307
308 vkprintf (4, "%s. in_total_bytes = %d\n", __func__, c->in.total_bytes);
309 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
310 int len;
311
312 while (1) {
313 if (c->flags & C_ERROR) {
314 return NEED_MORE_BYTES;
315 }
316 if (c->flags & C_STOPPARSE) {
317 return NEED_MORE_BYTES;
318 }
319 len = c->in.total_bytes;
320 if (len <= 0) {
321 return NEED_MORE_BYTES;
322 }
323
324 if (len < 4) {
325 return 4 - len;
326 }
327
328 int packet_len;
329 assert (rwm_fetch_lookup (&c->in, &packet_len, 4) == 4);
330
331 if (D->crypto_flags & RPCF_QUICKACK) {
332 D->flags = (D->flags & ~RPC_F_QUICKACK) | (packet_len & RPC_F_QUICKACK);
333 packet_len &= ~RPC_F_QUICKACK;
334 }
335
336 if (packet_len <= 0 || (packet_len & 0xc0000003)) {
337 if (D->in_packet_num <= -2 && (packet_len == *(int *)"HEAD" || packet_len == *(int *)"POST" || packet_len == *(int *)"GET " || packet_len == *(int *)"OPTI") && TCP_RPCS_FUNC(C)->http_fallback_type) {
338 vkprintf (1, "switching to http fallback for connection %d\n", c->fd);
339 memset (c->custom_data, 0, sizeof (c->custom_data));
340 c->type = TCP_RPCS_FUNC(C)->http_fallback_type;
341 c->extra = TCP_RPCS_FUNC(C)->http_fallback_extra;
342
343 if (c->type->init_accepted (C) < 0) {
344 vkprintf (1, "http init_accepted() returns error for connection %d\n", c->fd);
345 fail_connection (C, -33);
346 return 0;
347 }
348 //nbit_set (&c->Q, &c->In);
349 return c->type->parse_execute (C);
350 }
351 vkprintf (1, "error while parsing packet: bad packet length %d\n", packet_len);
352 fail_connection (C, -1);
353 return 0;
354 }
355
356 if ((packet_len > TCP_RPCS_FUNC(C)->max_packet_len && TCP_RPCS_FUNC(C)->max_packet_len > 0)) {
357 vkprintf (1, "error while parsing packet: bad packet length %d\n", packet_len);
358 fail_connection (C, -1);
359 return 0;
360 }
361
362 if (packet_len == 4) {
363 assert (rwm_skip_data (&c->in, 4) == 4);
364 continue;
365 }
366
367 if (packet_len < 16) {
368 vkprintf (1, "error while parsing packet: bad packet length %d\n", packet_len);
369 fail_connection (C, -1);
370 return 0;
371 }
372
373 if (len < packet_len) {
374 return packet_len - len;
375 }
376
377 struct raw_message msg;
378 rwm_split_head (&msg, &c->in, packet_len);
379
380 unsigned crc32;
381 assert (rwm_fetch_data_back (&msg, &crc32, 4) == 4);
382
383 unsigned packet_crc32 = rwm_custom_crc32 (&msg, packet_len - 4, D->custom_crc_partial);
384 if (crc32 != packet_crc32) {
385 vkprintf (1, "error while parsing packet: crc32 = %08x != %08x\n", packet_crc32, crc32);
386 rwm_dump (&msg);
387 fail_connection (C, -1);
388 rwm_free (&msg);
389 return 0;
390 }
391
392 int packet_num;
393 int packet_type;
394 assert (rwm_skip_data (&msg, 4) == 4);
395 assert (rwm_fetch_data (&msg, &packet_num, 4) == 4);
396 assert (rwm_fetch_lookup (&msg, &packet_type, 4) == 4);
397 packet_len -= 12;
398
399 if (verbosity > 2) {
400 fprintf (stderr, "received packet from connection %d (num %d)\n", c->fd, packet_num);
401 rwm_dump (&msg);
402 }
403
404 int res = -1;
405
406 if (D->in_packet_num == -3) {
407 D->in_packet_num = 0;
408 }
409
410 if (!(D->crypto_flags & RPCF_SEQNO_HOLES) && packet_num != D->in_packet_num) {
411 vkprintf (1, "error while parsing packet: got packet num %d, expected %d\n", packet_num, D->in_packet_num);
412 fail_connection (C, -1);
413 rwm_free (&msg);
414 return 0;
415 } else if (packet_num < 0) {
416 /* this is for us */
417 if (packet_num == -2) {
418 res = tcp_rpcs_process_nonce_packet (C, &msg); // if res > 0, nonce packet sent in response
419 } else if (packet_num == -1) {
420 res = tcp_rpcs_process_handshake_packet (C, &msg);
421 if (res >= 0) {
422 res = tcp_rpcs_send_handshake_packet (C);
423 if (D->crypto_flags & RPCF_USE_CRC32C) {
424 D->custom_crc_partial = crc32c_partial;
425 }
426 notification_event_insert_tcp_conn_ready (C);
427 }
428 }
429 rwm_free (&msg);
430 if (res < 0) {
431 fail_connection (C, res);
432 return 0;
433 }
434 } else {
435 /* main case */
436 c->last_response_time = precise_now;
437 if (packet_type == RPC_PING) {
438 res = tcp_rpcs_default_execute (C, packet_type, &msg);
439 } else {
440 res = TCP_RPCS_FUNC(C)->execute (C, packet_type, &msg);
441 }
442 if (res <= 0) {
443 rwm_free (&msg);
444 }
445 }
446
447 D->in_packet_num++;
448 }
449 return NEED_MORE_BYTES;
450}

494int tcp_rpcs_init_accepted (connection_job_t C) {
495 struct connection_info *c = CONN_INFO (C);
496
497 c->last_query_sent_time = precise_now;
498 TCP_RPC_DATA(C)->custom_crc_partial = crc32_partial;
499
500 if (TCP_RPCS_FUNC(C)->rpc_check_perm) {
501 int res = TCP_RPCS_FUNC(C)->rpc_check_perm (C);
502 vkprintf (4, "tcp_rpcs_check_perm for connection %d: [%s]:%d -> [%s]:%d = %d\n", c->fd, show_remote_ip (C), c->remote_port, show_our_ip (C), c->our_port, res);
503 if (res < 0) {
504 return res;
505 }
506 res &= RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC | RPCF_REQ_DH | RPCF_ALLOW_SKIP_DH;
507 if (!(res & (RPCF_ALLOW_UNENC | RPCF_ALLOW_ENC))) {
508 return -1;
509 }
510
511 TCP_RPC_DATA(C)->crypto_flags = res;
512 } else {
513 TCP_RPC_DATA(C)->crypto_flags = RPCF_ALLOW_UNENC;
514 }
515
516 TCP_RPC_DATA(C)->in_packet_num = -2;
517 TCP_RPC_DATA(C)->out_packet_num = -2;
518
519 return 0;
520}

548}

557int tcp_rpcs_init_crypto (connection_job_t C, struct tcp_rpc_nonce_packet *P) {
558 struct connection_info *c = CONN_INFO (C);
559
560// fprintf (stderr, "rpcs_init_crypto (%p [fd=%d], '%.*s')\n", c, c->fd, key_len, key);
561 struct tcp_rpc_data *D = TCP_RPC_DATA(C);
562
563 if (c->crypto) {
564 return -1;
565 }
566
567 if ((D->crypto_flags & (RPCF_ALLOW_ENC | RPCF_ALLOW_UNENC)) == RPCF_ALLOW_UNENC) {
568 return tcp_rpcs_init_fake_crypto (C);
569 }
570
571 if ((D->crypto_flags & (RPCF_ALLOW_ENC | RPCF_ALLOW_UNENC)) != RPCF_ALLOW_ENC) {
572 return -1;
573 }
574
575 if (main_secret.key_signature != P->key_select) {
576 return -1;
577 }
578
579 aes_secret_t *secret = &main_secret;
580
581 union {
582 struct tcp_rpc_nonce_packet s;
583 struct tcp_rpc_nonce_ext_packet x;
584 struct tcp_rpc_nonce_dh_packet dh;
585 } buf;
586
587 struct tcp_rpc_nonce_dh_packet *old_dh = 0, *new_dh = 0;
588 unsigned char temp_dh[256];
589 int temp_dh_len = 0;
590
591 if (D->crypto_flags & RPCF_REQ_DH) {
592 new_dh = (struct tcp_rpc_nonce_dh_packet *)((char *)&buf - 4*RPC_MAX_EXTRA_KEYS);
593 if (P->crypto_schema != RPC_CRYPTO_AES_DH) {
594 return -1;
595 }
596 old_dh = (struct tcp_rpc_nonce_dh_packet *)((char *)P + 4*(((struct tcp_rpc_nonce_dh_packet *) P)->extra_keys_count - RPC_MAX_EXTRA_KEYS));
597 if (old_dh->dh_params_select != dh_params_select || !dh_params_select) {
598 return -1;
599 }
600
601 if (tcp_add_dh_accept () < 0) {
602 return -1;
603 }
604
605 temp_dh_len = dh_second_round (temp_dh, new_dh->g_a, old_dh->g_a);
606 assert (temp_dh_len == 256);
607
608 incr_active_dh_connections ();
609 __sync_fetch_and_or (&c->flags, C_ISDH);
610 }
611 aes_generate_nonce (D->nonce);
612
613 struct aes_key_data aes_keys;
614
615 if (aes_create_keys (&aes_keys, 0, D->nonce, P->crypto_nonce, P->crypto_ts, nat_translate_ip (c->our_ip), c->our_port, c->our_ipv6, nat_translate_ip (c->remote_ip), c->remote_port, c->remote_ipv6, secret, temp_dh, temp_dh_len) < 0) {
616 return -1;
617 }
618
619 if (aes_crypto_init (C, &aes_keys, sizeof (aes_keys)) < 0) {
620 return -1;
621 }
622
623 memcpy (buf.s.crypto_nonce, D->nonce, 16);
624 buf.s.crypto_ts = D->nonce_time;
625 buf.s.type = RPC_NONCE;
626 buf.s.key_select = secret->key_signature;
627
628 int buf_len;
629 if (!new_dh) {
630 buf.s.crypto_schema = RPC_CRYPTO_AES;
631 buf_len = sizeof (struct tcp_rpc_nonce_packet);
632 } else {
633 buf.dh.crypto_schema = RPC_CRYPTO_AES_DH;
634 buf_len = sizeof (struct tcp_rpc_nonce_dh_packet) - 4*RPC_MAX_EXTRA_KEYS;
635 buf.dh.extra_keys_count = 0;
636 new_dh->dh_params_select = dh_params_select;
637 }
638
639 assert ((D->crypto_flags & (RPCF_ALLOW_ENC | RPCF_ENC_SENT)) == RPCF_ALLOW_ENC);
640 D->crypto_flags |= RPCF_ENC_SENT;
641
642 tcp_rpc_conn_send_data_init (C, buf_len, &buf);
643
644
645 return 1;
646}
